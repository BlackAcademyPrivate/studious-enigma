local activeSlider = nil

local function GetGuiParent()
    if type(gethui) == "function" then
        local ok, result = pcall(gethui)
        if ok and result then return result end
    end

    local ok, coreGui = pcall(function() return game:GetService("CoreGui") end)
    if ok and coreGui then return coreGui end

    return Players.LocalPlayer:WaitForChild("PlayerGui")
end

local function Tween(instance, props, t)
    TweenService:Create(instance, TweenInfo.new(t or 0.15, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), props):Play()
end

local function New(class, props)
    local inst = Instance.new(class)
    for k, v in pairs(props) do
        inst[k] = v
    end
    return inst
end

local function Corner(parent, px)
    local c = Instance.new("UICorner")
    c.CornerRadius = UDim.new(0, px or 6)
    c.Parent = parent
    return c
end

local function Stroke(parent, color, thickness)
    local s = Instance.new("UIStroke")
    s.Color     = color or Color3.fromRGB(40, 40, 40)
    s.Thickness = thickness or 1
    s.Parent    = parent
    return s
end

local function ListLayout(parent, padding, dir)
    local l = Instance.new("UIListLayout")
    l.Padding       = UDim.new(0, padding or 0)
    l.FillDirection = dir or Enum.FillDirection.Vertical
    l.SortOrder     = Enum.SortOrder.LayoutOrder
    l.Parent        = parent
    return l
end

local function Padding(parent, top, bottom, left, right)
    local p = Instance.new("UIPadding")
    p.PaddingTop    = UDim.new(0, top    or 0)
    p.PaddingBottom = UDim.new(0, bottom or 0)
    p.PaddingLeft   = UDim.new(0, left   or 0)
    p.PaddingRight  = UDim.new(0, right  or 0)
    p.Parent        = parent
    return p
end

local function IsMouse1(input)
    return input.UserInputType == Enum.UserInputType.MouseButton1
end

local function IsMouseMove(input)
    return input.UserInputType == Enum.UserInputType.MouseMovement
        or input.UserInputType == Enum.UserInputType.Touch
end

local function IsTouch(input)
    return input.UserInputType == Enum.UserInputType.Touch
end

local function MakeDraggable(handle, target)
    local dragging  = false
    local dragStart = Vector2.zero
    local startPos  = UDim2.new()

    handle.InputBegan:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            dragging  = true
            dragStart = Vector2.new(input.Position.X, input.Position.Y)
            startPos  = target.Position
            input:GetPropertyChangedSignal("UserInputState"):Connect(function()
                if input.UserInputState == Enum.UserInputState.End then
                    dragging = false
                end
            end)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if dragging and IsMouseMove(input) then
            local delta = Vector2.new(input.Position.X, input.Position.Y) - dragStart
            target.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X,
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if IsMouse1(input) or IsTouch(input) then
            dragging = false
        end
    end)
end

local ElementBuilders = {}

local function BaseElement(parent, height)
    local frame = New("Frame", {
        BackgroundColor3 = Theme.ElementBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, height or S.ElemH),
        Parent           = parent,
    })
    Corner(frame, 4)

    frame.MouseEnter:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ElementHover }) end)
    frame.MouseLeave:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ElementBg }) end)

    return frame
end

function ElementBuilders.Button(parent, opts)
    local label    = opts.Title    or "Button"
    local callback = opts.Callback or function() end

    local frame = New("Frame", {
        BackgroundColor3 = Theme.ButtonBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, S.ElemH),
        Parent           = parent,

    Corner(frame, 4)

    frame.MouseEnter:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ButtonHover }) end)
    frame.MouseLeave:Connect(function() Tween(frame, { BackgroundColor3 = Theme.ButtonBg }) end)

        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 14,
        TextXAlignment         = Enum.TextXAlignment.Center,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Position               = UDim2.new(0, 0, 0, 0),
        Parent                 = frame,
    })

    local btn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = frame,
    })

    btn.MouseButton1Click:Connect(function()
        Tween(frame, { BackgroundColor3 = Theme.AccentDark }, 0.08)
        task.delay(0.14, function()
            Tween(frame, { BackgroundColor3 = Theme.ButtonBg }, 0.12)
        end)
        pcall(callback)
    end)

function ElementBuilders.Toggle(parent, opts)
    local label    = opts.Title    or "Toggle"
    local state    = opts.Default  or false
    local callback = opts.Callback or function() end

    local frame = BaseElement(parent)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 14,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -50, 1, 0),
        Position               = UDim2.new(0, 14, 0, 0),
        Parent                 = frame,
    })

    local BOX = 18
    local box = New("Frame", {
        BackgroundColor3 = state and Theme.CheckOn or Theme.ButtonBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(0, BOX, 0, BOX),
        Position         = UDim2.new(1, -(BOX + 10), 0.5, -(BOX / 2)),
        Parent           = frame,
    })
    Corner(box, 3)

    local mark = New("ImageLabel", {
        Image                  = "rbxassetid://10709790644",
        ImageColor3            = Theme.CheckMark,
        ImageTransparency      = state and 0 or 1,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -4, 1, -4),
        Position               = UDim2.new(0, 2, 0, 2),
        Parent                 = box,
    })

    local btn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = frame,
    })

    local function Toggle()
        state = not state
        Tween(box,  { BackgroundColor3 = state and Theme.CheckOn or Theme.ButtonBg }, 0.13)
        Tween(mark, { ImageTransparency = state and 0 or 1 }, 0.10)
        pcall(callback, state)
    end

    btn.MouseButton1Click:Connect(Toggle)

    local api = {}
    function api:SetState(v)
        state = v
        box.BackgroundColor3   = v and Theme.CheckOn or Theme.ButtonBg
        mark.ImageTransparency = v and 0 or 1
    end
    function api:GetState() return state end
    return api
end

function ElementBuilders.Slider(parent, opts, contentArea)
    local label    = opts.Title    or "Slider"
    local minVal   = opts.Min      or 0
    local maxVal   = opts.Max      or 100
    local default  = math.clamp(opts.Default or minVal, minVal, maxVal)
    local suffix   = opts.Suffix   or ""
    local callback = opts.Callback or function() end

    local frame = BaseElement(parent, S.ElemH + 14)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 14,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -65, 0, 20),
        Position               = UDim2.new(0, 14, 0, 5),
        Parent                 = frame,
    })

    local valLabel = New("TextLabel", {
        Text                   = tostring(default) .. suffix,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.Font,
        TextSize               = 13,
        TextXAlignment         = Enum.TextXAlignment.Right,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0, 55, 0, 20),
        Position               = UDim2.new(1, -62, 0, 5),
        Parent                 = frame,
    })

    local PX = 14
    local track = New("Frame", {
        BackgroundColor3 = Theme.SliderTrack,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, -(PX * 2), 0, 5),
        Position         = UDim2.new(0, PX, 0, 29),
        Parent           = frame,
    })
    Corner(track, 3)

    local initPct = (default - minVal) / (maxVal - minVal)
    local fill = New("Frame", {
        BackgroundColor3 = Theme.SliderFill,
        BorderSizePixel  = 0,
        Size             = UDim2.new(initPct, 0, 1, 0),
        Parent           = track,
    })
    Corner(fill, 3)

    local KS = 13
    local knob = New("Frame", {
        BackgroundColor3 = Theme.SliderKnob,
        BorderSizePixel  = 0,
        Size             = UDim2.new(0, KS, 0, KS),
        Position         = UDim2.new(initPct, -(KS / 2), 0.5, -(KS / 2)),
        Parent           = track,
    })
    Corner(knob, KS)

    local current  = default
    local sliderId = {}

    local function Update(absX)
        local ap  = track.AbsolutePosition
        local as  = track.AbsoluteSize
        local pct = math.clamp((absX - ap.X) / as.X, 0, 1)
        local val = math.round(minVal + (maxVal - minVal) * pct)
        fill.Size     = UDim2.new(pct, 0, 1, 0)
        knob.Position = UDim2.new(pct, -(KS / 2), 0.5, -(KS / 2))
        if val ~= current then
            current = val
            valLabel.Text = tostring(val) .. suffix
            pcall(callback, val)
        end
    end

    track.InputBegan:Connect(function(input)
        if (IsMouse1(input) or IsTouch(input)) and activeSlider == nil then
            activeSlider = sliderId
            if contentArea then contentArea.ScrollingEnabled = false end
            Update(input.Position.X)
        end
    end)

    UserInputService.InputChanged:Connect(function(input)
        if activeSlider == sliderId and IsMouseMove(input) then
            Update(input.Position.X)
        end
    end)

    UserInputService.InputEnded:Connect(function(input)
        if activeSlider == sliderId and (IsMouse1(input) or IsTouch(input)) then
            activeSlider = nil
            if contentArea then contentArea.ScrollingEnabled = true end
        end
    end)

    local api = {}
    function api:SetValue(v)
        v = math.clamp(math.round(v), minVal, maxVal)
        local pct = (v - minVal) / (maxVal - minVal)
        fill.Size     = UDim2.new(pct, 0, 1, 0)
        knob.Position = UDim2.new(pct, -(KS / 2), 0.5, -(KS / 2))
        valLabel.Text = tostring(v) .. suffix
        current = v
    end
    function api:GetValue() return current end
    return api
end

function ElementBuilders.Section(parent, opts, contentArea)
    local label     = opts.Title   or "Section"
    local collapsed = opts.Default == true

    local wrapper = New("Frame", {
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        Size                   = UDim2.new(1, 0, 0, S.ElemH),
        ClipsDescendants       = true,
        Parent                 = parent,
    })

    local header = New("Frame", {
        BackgroundColor3 = Theme.ElementBg,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, S.ElemH),
        Parent           = wrapper,
    })
    Corner(header, 4)

    header.MouseEnter:Connect(function() Tween(header, { BackgroundColor3 = Theme.ElementHover }) end)
    header.MouseLeave:Connect(function() Tween(header, { BackgroundColor3 = Theme.ElementBg }) end)

    New("TextLabel", {
        Text                   = label,
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 14,
        TextXAlignment         = Enum.TextXAlignment.Left,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, -30, 1, 0),
        Position               = UDim2.new(0, 14, 0, 0),
        Parent                 = header,
    })

    local iconLabel = New("TextLabel", {
        Text                   = collapsed and "▶" or "▼",
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 20,
        TextXAlignment         = Enum.TextXAlignment.Center,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(0, 26, 1, 0),
        Position               = UDim2.new(1, -30, 0, 0),
        Parent                 = header,
    })

    local dividerHolder = New("Frame", {
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        Size                   = UDim2.new(1, 0, 0, 6),
        Position               = UDim2.new(0, 0, 0, S.ElemH),
        Visible                = not collapsed,
        Parent                 = wrapper,
    })
    TitleDivider(dividerHolder)

    local body = New("Frame", {
        BackgroundTransparency = 1,
        BorderSizePixel        = 0,
        Size                   = UDim2.new(1, 0, 0, 0),
        Position               = UDim2.new(0, 0, 0, S.ElemH + 6),
        ClipsDescendants       = true,
        Parent                 = wrapper,
    })

    local bodyLayout = ListLayout(body, S.ElemPad)

    local function RecalcSizes(animate)
        local h = bodyLayout.AbsoluteContentSize.Y
        local targetBodyH = collapsed and 0 or h
        local targetWrapH = collapsed and S.ElemH or S.ElemH + 6 + h
        if animate then
            Tween(body,    { Size = UDim2.new(1, 0, 0, targetBodyH) }, 0.15)
            Tween(wrapper, { Size = UDim2.new(1, 0, 0, targetWrapH) }, 0.15)
        else
            body.Size    = UDim2.new(1, 0, 0, targetBodyH)
            wrapper.Size = UDim2.new(1, 0, 0, targetWrapH)
        end
        if contentArea then
            task.defer(function()
                local l = contentArea:FindFirstChildWhichIsA("UIListLayout")
                if l then
                    contentArea.CanvasSize = UDim2.new(0, 0, 0, l.AbsoluteContentSize.Y + S.SidePad * 2)
                end
            end)
        end
    end

    bodyLayout:GetPropertyChangedSignal("AbsoluteContentSize"):Connect(function() RecalcSizes(false) end)

    local headerBtn = New("TextButton", {
        Text                   = "",
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Parent                 = header,
    })

    headerBtn.MouseButton1Click:Connect(function()
        collapsed             = not collapsed
        iconLabel.Text        = collapsed and "▶" or "▼"
        dividerHolder.Visible = not collapsed
        RecalcSizes(true)
    end)

    RecalcSizes(false)

    return body, contentArea
end

        BorderSizePixel  = 0,
        Size             = UDim2.new(0, S.W, 0, S.H),
        Position         = UDim2.new(0.5, -(S.W / 2), 0.5, -(S.H / 2)),

    })

    local titleBar = New("Frame", {
        Name             = "TitleBar",
        BackgroundColor3 = Theme.TitleBar,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, S.TitleH),
        Parent           = mainFrame,
    })
    Corner(titleBar, 6)

    local titleBarFiller = New("Frame", {
        BackgroundColor3 = Theme.TitleBar,
        BorderSizePixel  = 0,
        Size             = UDim2.new(1, 0, 0, 10),
        Position         = UDim2.new(0, 0, 1, -10),
        Parent           = titleBar,
    })

    New("TextLabel", {
        Text                   = title,
        TextColor3             = Theme.TitleText,
        Font                   = Theme.Font,
        TextSize               = 14,
        TextXAlignment         = Enum.TextXAlignment.Center,
        BackgroundTransparency = 1,
        Size                   = UDim2.new(1, 0, 1, 0),
        Position               = UDim2.new(0, 0, 0, 0),
        Parent                 = titleBar,
    })

    local windowCollapseIcon = New("TextLabel", {
        Text                   = "▼",
        TextColor3             = Theme.ElementText,
        Font                   = Theme.FontSemi,
        TextSize               = 20,
        TextXAlignment         = Enum.TextXAlignment.Center,

        Position               = UDim2.new(1, -30, 0, 0),

    })

        Text                   = "",

        contentArea.CanvasSize = UDim2.new(0, 0, 0, layout.AbsoluteContentSize.Y + S.SidePad * 2)
    end)

    local self             = setmetatable({}, Window)
    self._Gui              = screenGui
    self._Content          = contentArea
    self._SectionBody      = nil
    self._SectionRemaining = 0

    return self
end

local function ResolveTarget(win)
    if win._SectionRemaining > 0 then
        win._SectionRemaining = win._SectionRemaining - 1
        if win._SectionRemaining == 0 then
            local body = win._SectionBody
            win._SectionBody      = nil
            win._SectionRemaining = 0
            return body
        end
        return win._SectionBody
    end
    return win._Content
end

function Window:AddButton(opts)
    ElementBuilders.Button(ResolveTarget(self), opts)
end

function Window:AddToggle(opts)
    return ElementBuilders.Toggle(ResolveTarget(self), opts)
end

function Window:AddSlider(opts)
    local target = ResolveTarget(self)
    return ElementBuilders.Slider(target, opts, self._Content)
end

function Window:AddInput(opts)
    return ElementBuilders.Input(ResolveTarget(self), opts)
end

function Window:AddSection(opts)
    local count = opts.Count or 1
    local body, ca = ElementBuilders.Section(self._Content, opts, self._Content)
    self._SectionBody      = body
    self._SectionRemaining = count
end

function Window:Destroy()
    if self._Gui then self._Gui:Destroy() end
end
